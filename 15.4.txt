SELECT i.ID, i.SUMMARY, i.DESCRIPTION, il.NAME AS LIST_NAME
FROM issues i
JOIN ISSUESLIST il ON i.ISSUESLIST_ID = il.ID;

2.
SELECT i.ID, i.SUMMARY, i.DESCRIPTION, u.FIRSTNAME, u.LASTNAME AS LIST_NAME
FROM ISSUES i
JOIN USERS u ON i.USER_ID_ASSIGNEDTO = u.ID;

3.
SELECT u.FIRSTNAME, u.LASTNAME, COUNT(i.ID) AS TASK_COUNT
FROM USERS U
JOIN ISSUES i ON u.ID = i.USER_ID_ASSIGNEDTO
GROUP BY u.FIRSTNAME, u.LASTNAME
ORDER BY TASK_COUNT DESC;
